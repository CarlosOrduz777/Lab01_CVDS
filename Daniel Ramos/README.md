def sum(num_1, num_2):
    resultado = num_1 + num_2
    return resultado
def mult(num_1, num_2):
    resultado = 0
    for i in range (num_2):
        resultado = sum(resultado, num_1)
    return resultado
def pow(num_1, num_2):
    resultado = 1
    for i in range (num_2):
        resultado = mult(resultado, num_1)
    return  resultado
def main():
    #print("Por favor digite el primer numero")
